create table user(
   Id not null AUTO_INCREMENT,
   Match_Id numeric,	
   Status char,
   Type char,
   Gender char,
   Zip varchar(5),
   Borough varchar(10),
   E-mail varchar(255),
   Phone varchar(255),
   Employed boolean,
   Age integer,
   Married boolean,
   Interests varchar(255),
   Preferred_Communication varchar(255),
   BestTimes_To_Socialize varchar(255),
   Travel_Independent boolean,
   PRIMARY KEY (Id)
)
