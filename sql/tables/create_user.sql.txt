create table user(
   Id int not null AUTO_INCREMENT,
   Match_Id int,
   Name varchar(25),
   Status char,
   Type char,
   Gender char,
   Zip varchar(5),
   Borough varchar(10),
   Email varchar(25),
   Phone varchar(10),
   Employed boolean,
   Age int,
   Married boolean,
   Interests varchar(255),
   Preferred_Communication varchar(255),
   BestTimes_To_Socialize varchar(255),
   Travel_Independent boolean,
   Days_Since_Matched int,
   About_Me varchar(255),
   PRIMARY KEY (Id)
);
